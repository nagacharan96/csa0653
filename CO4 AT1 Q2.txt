    def kruskal(edges, n):
    parent = list(range(n))
    def find(x):
        if parent[x] != x:
            parent[x] = find(parent[x])
        return parent[x]
    def union(x, y):
        parent[find(x)] = find(y)
    edges.sort(key=lambda x: x[2])
    mst_weight = 0
    mst_edges = []
  for u, v, w in edges:
        if find(u) != find(v):
            union(u, v)
            mst_edges.append((u, v, w))
            mst_weight += w
    return mst_edges, mst_weight
edges = [
    (0, 1, 2),
    (0, 2, 3),
    (1, 2, 1),
    (1, 3, 4),
    (2, 3, 5)
]
mst, weight = kruskal(edges, 4)
print("Edges in MST:")
for u, v, w in mst:
    print(f"{u} - {v} : {w}")

print("Total Weight =", weight)
